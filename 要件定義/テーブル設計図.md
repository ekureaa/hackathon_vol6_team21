    USERS {
        SERIAL id PK "ユーザーID"
        VARCHAR email "メールアドレス"
        VARCHAR password_hash "パスワードハッシュ"
        VARCHAR username "ユーザー名"
        VARCHAR profile_image_url "プロフィール画像のURL"
        TIMESTAMP created_at "アカウント作成日時"
        TIMESTAMP updated_at "アカウント更新日時"
    }

    IMAGES {
        SERIAL id PK "画像ID"
        INTEGER user_id FK "ユーザーID"
        VARCHAR image_url "画像のURL"
        TIMESTAMP uploaded_at "アップロード日時"
        BOOLEAN is_deleted "削除フラグ"
    }

    IMAGE_RECOGNITIONS {
        SERIAL id PK "画像認識ID"
        INTEGER image_id FK "画像ID"
        VARCHAR recognized_dish "認識された料理名"
        DECIMAL confidence_level "認識の信頼度"
        VARCHAR feedback "ユーザーからのフィードバック"
        TIMESTAMP created_at "認識日時"
    }
    RECIPE_SUGGESTIONS {
        SERIAL id PK "レシピ提案ID"
        INTEGER recognition_id FK "画像認識ID"
        INTEGER recipe_id FK "レシピID"
        TIMESTAMP created_at "提案日時"
        BOOLEAN is_public "公開フラグ"
    }


    USERS ||--o{ IMAGES : "has"
    IMAGES ||--o{ IMAGE_RECOGNITIONS : "is"
    IMAGE_RECOGNITIONS ||--o{ RECIPE_SUGGESTIONS : "is"
